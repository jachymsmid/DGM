24// ─────────────────────────────────────────────────────────────────────────────
25// writeToVTK
26//
27// Writes the DG solution to a legacy ASCII VTK file that Paraview can open.
28//
29// Strategy: the DG solution lives on sub-element nodes, not on the mesh
30// vertices. We therefore write a *refined* point cloud: for each element k
31// we emit its Np physical node positions as VTK points, and connect each
32// group of Np points into a VTK_POLY_LINE cell. This gives Paraview a
33// piecewise-polynomial picture of u inside every element, with the expected
34// discontinuities visible at element boundaries.
35//
36// Output layout:
37// - K * Np points (physical DG node positions)
38// - K cells (one VTK_POLY_LINE per element, with Np vertices)
39// - 1 point data array named after `fieldName`
40//
41// Usage:
42// DG::writeToVTK(mesh, ref, u, "output.vtk", "u");
43// ─────────────────────────────────────────────────────────────────────────────
44template
45<
46 class Real = double,
47 class Device = TNL::Devices::Host,
48 class Index = int
49>
58void writeToVTK(const Mesh<Real, Device, Index>& mesh,
59 const ReferenceElement<Real, Index>& ref,
60 const FieldVector<Real, Device, Index>& u,
61 const std::string& filename,
62 const std::string& fieldName = "u",
63 Real t = Real(0))
64{
65 const Index K = mesh.numElements();
66 const Index Np = ref.numDOF();
67 const Index totalPts = K * Np;
68
69 std::ofstream f(filename);
70 if (!f)
71 throw std::runtime_error("writeToVTK: cannot open '" + filename + "'");
72
73 f << std::scientific << std::setprecision(10);
74
75 // ── VTK header ──────────────────────────────────────────────────────────
76 f << "# vtk DataFile Version 3.0\n";
77 f << "DG1D solution: " << fieldName << " t=" << t << "\n";
78 f << "ASCII\n";
79 f << "DATASET UNSTRUCTURED_GRID\n";
80
81 // ── Points ──────────────────────────────────────────────────────────────
82 // VTK always needs 3D coordinates; pad y=0, z=0 for 1D.
83 f << "POINTS " << totalPts << " double\n";
84 for (Index k = 0; k < K; ++k)
85 {
86 const Real xL = mesh.leftVertex(k);
87 const Real h = mesh.elementSize(k);
88 for (Index i = 0; i < Np; ++i)
89 {
90 // map reference node r_i in [-1,1] to physical coordinate
91 const Real r = ref.nodes()[i];
92 const Real x = xL + (r + Real(1)) * Real(0.5) * h;
93 f << x << " 0.0 0.0\n";
94 }
95 }
96
97 // ── Cells ────────────────────────────────────────────────────────────────
98 // Each cell is a VTK_POLY_LINE with Np vertices.
99 // Cell list entry: [count, v0, v1, ..., v_{Np-1}]
100 const Index cellListSize = K * (Np + 1); // Np vertices + 1 count per cell
101 f << "\nCELLS " << K << " " << cellListSize << "\n";
102 for (Index k = 0; k < K; ++k)
103 {
104 f << Np;
105 for (Index i = 0; i < Np; ++i)
106 {
107 f << " " << k * Np + i;
108 }
109 f << "\n";
110 }
111
112 // VTK_POLY_LINE = cell type 4
113 f << "\nCELL_TYPES " << K << "\n";
114 for (Index k = 0; k < K; ++k)
115 {
116 f << "4\n";
117 }
118
119 // ── Point data ───────────────────────────────────────────────────────────
120 f << "\nPOINT_DATA " << totalPts << "\n";
121 f << "SCALARS " << fieldName << " double 1\n";
122 f << "LOOKUP_TABLE default\n";
123 for (Index k = 0; k < K; ++k)
124 {
125 const Real* uk = u.elementPtr(k);
126 for (Index i = 0; i < Np; ++i)
127 {
128 f << uk[i] << "\n";
129 }
130 }
131}

133// ─────────────────────────────────────────────────────────────────────────────
134// writeTimeSeriesVTK
135//
136// Convenience wrapper: writes a numbered sequence of VTK files suitable for
137// loading as a time series in Paraview.
138//
139// Call it from your RK4 callback:
140//
141// int frame = 0;
142// auto cb = [&](Real t, const Field& u, int step) -> bool {
143// if (step % 10 == 0)
144// DG::writeTimeSeriesVTK(mesh, ref, u, "output", frame++, t);
145// return true;
146// };
147//
148// Produces: output_000000.vtk, output_000001.vtk, ...
149// Load them in Paraview with File > Open, select all, apply "Group Datasets".
150// ─────────────────────────────────────────────────────────────────────────────
151template
152<
153 class Real,
154 class Device,
155 class Index
156>
163void writeTimeSeriesVTK(const Mesh<Real, Device, Index>& mesh,
164 const ReferenceElement<Real, Index>& ref,
165 const FieldVector<Real, Device, Index>& u,
166 const std::string& basename,
167 int frameIndex,
168 Real t,
169 const std::string& fieldName = "u")
170{
171 // Zero-pad frame index to 6 digits: output_000042.vtk
172 std::ostringstream oss;
173 oss << basename << "_" << std::setfill('0') << std::setw(6) << frameIndex << ".vtk";
174 writeToVTK(mesh, ref, u, oss.str(), fieldName, t);
175}
