52 // Factory: uniform mesh on [a,b] with K elements
60 static Mesh uniform(Real a, Real b, Index K)
61 {
62 if (K < 1) throw std::invalid_argument("K must be >= 1");
63 using Builder = TNL::Meshes::MeshBuilder<TNLMeshType>;
64 Builder builder;
65 builder.setEntitiesCount(K + 1, K); // K+1 vertices, K cells
66
67 const Real h = (b - a) / K;
68
69 for (Index i = 0; i <= K; ++i)
70 builder.setPoint(i, TNL::Containers::StaticVector<1,Real>{a + i * h});
71
72 for (Index k = 0; k < K; ++k) {
73 builder.getCellSeed(k).setCornerId(0, k);
74 builder.getCellSeed(k).setCornerId(1, k + 1);
75 }
76 TNLMeshType m;
77 builder.build(m);
78 return Mesh(std::move(m));
79 }

173 void buildDerivedData_()
174 {
175 // number of cells in a TNL mesh
176 numK_ = mesh_.template getEntitiesCount<1>();
177
178 // number of vertices
179 vertCoords_.setSize(numK_ + 1);
180
181 Real h = 0;
182 Real vertCoordPrev;
183
184 // for every mesh cell
185 // set coordinates of the vertices, get max and min spatial step
186 for (Index i = 0; i <= numK_; ++i)
187 {
188 auto pt = mesh_.template getEntity<0>(i).getPoint();
189 vertCoords_[i] = pt[0];
190 if (i == 0)
191 {
192 vertCoordPrev = pt[0];
193 continue;
194 }
195 h = vertCoords_[i] - vertCoordPrev;
196 max_h_ = std::max(h, max_h_);
197 min_h_ = std::min(h, min_h_);
198 vertCoordPrev = pt[0];
199 }
200 }
