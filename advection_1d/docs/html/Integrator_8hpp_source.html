42class Integrator
43{
44public:
45 using Field = FieldVector<Real, Device, Index>;
46 using RHSFunc = std::function<void(const Field&, Field&, const Real&)>;
47 using Callback = StepCallback<Real, Device, Index>;
48

50
51 // single step
58 virtual void step(Field& u, Real dt, Real t_in) = 0;
59
60 // integrate, multiple steps
65 virtual Index integrate(Field& u, Real t0, Real t_end, Real dt) = 0;
66

82 // scale a field by a scalar
83 void scaleField_(Field& f, Real s) const
84 {
85 const Index total = numPoints();
86 Real* data = f.data().getData();
87 for (Index i = 0; i < total; ++i)
88 {
89 data[i] *= s;
90 }
91 }
92
93 // res = base + alpha * delta
94 void addScaled_(const Field& base,
95 const Field& delta,
96 Real alpha,
97 Field& out) const
98 {
99 const Index total = numPoints();
100 const Real* b = base.data().getData();
101 const Real* d = delta.data().getData();
102 Real* o = out.data().getData();
103 for (Index i = 0; i < total; ++i)
104 {
105 o[i] = b[i] + alpha * d[i];
106 }
107 }

111// ------------------------------- ERK4 ---------------------------------------
112// simple explicit four stage Runge-Kutta
113template<typename Real = double,
114 typename Device = TNL::Devices::Host,
115 typename Index = int>
123class ERK : Integrator< Real, Device, Index > // inherit protected methods
124{
125public:
126 using Field = FieldVector<Real, Device, Index>;
127 using RHSFunc = std::function<void(const Field&, Field&, const Real&)>;
128 using Callback = StepCallback<Real, Device, Index>;
129
130 // op – spatial residual (must outlive this integrator)
131 // K, Np – element count and nodes-per-element (needed to size scratch)
132 // callback – optional: called after every step, return false to stop early
140 explicit ERK(RHSFunc rhs,
141 Index K, Index Np,
142 Callback callback = nullptr)
143 : rhs_(std::move(rhs)),
144 K_(K), Np_(Np),
145 callback_(std::move(callback)),
146 k1_(K, Np), k2_(K, Np), k3_(K, Np), k4_(K, Np),
147 tmp_(K, Np) {}
148
149 // single step
150 void step(Field& u, Real dt, Real t_in) override
151 {
152 if (dt <= Real(0))
153 throw std::invalid_argument("DG::Integrator::step: dt must be > 0");
154
155 // k1 = L(u^n, t_n)
156 rhs_(u, k1_, t_in + 0.5 * dt);
157
158 // k2 = L(u^n + 1/2 dt k1, t_n + 1/2 dt)
159 // tmp = u + 0.5*dt*k1
160 this -> addScaled_(u, k1_, 0.5 * dt, tmp_);
161 rhs_(tmp_, k2_, t_in + 0.5 * dt);
162
163 // k3 = L(u^n + 1/2 dt k2, t_n + 1/2 dt)
164 // tmp = u + 0.5*dt*k2
165 this -> addScaled_(u, k2_, 0.5 * dt, tmp_);
166 rhs_(tmp_, k3_, t_in + 0.5 * dt);
167
168 // k4 = L(u^n + dt k3, t_n + dt)
169 // tmp = u + dt * k3
170 this -> addScaled_(u, k3_, dt, tmp_);
171 rhs_(tmp_, k4_, t_in + dt );
172
173 // combine: u += (k1 + 2*k2 + 2*k3 + k4) / 6
174 combine_(u, k1_, k2_, k3_, k4_, dt);
175 }
176
177 // integrate: perform multiple steps
178 Index integrate(Field& u, Real t0, Real t_end, Real dt) override
179 {
180 if (t_end <= t0)
181 throw std::invalid_argument("Integrator::integrate: t_end must be > t0");
182
183 Real t = t0;
184 Index nstep = 0;
185
186 while (t < t_end - Real(1e-12))
187 {
188 // clip the last step to reach t_end exactly
189 Real dt_actual = std::min(dt, t_end - t);
190
191 step(u, dt_actual, t);
192 t += dt_actual;
193 nstep += 1;
194
195 // Invoke optional callback; stop early if it returns false
196 if (callback_ && !callback_(t, u, nstep))
197 break;
198 }
199
200 return nstep;
201 }
202
203 // max dt from the Hesthaven book
204 static Real computeDt(Real x_min, Real max_wave_speed, int poly_order, Real cfl = Real(0.4))
205 {
206 if (TNL::abs(max_wave_speed) <= Real(0))
207 throw std::invalid_argument("max_wave_speed must be positive");
208 return cfl * x_min / (TNL::abs(max_wave_speed) * ( 2 * poly_order + 1 ));
209 }
210
211 // getters
212 Index numSteps() const override { return lastStepCount_; }
213 Real currentTime() const override { return currentTime_; }
214 Index numPoints() const override { return K_ * Np_; }
215
216private:
217 // combine all the stages
218 void combine_(Field& u,
219 const Field& k1,
220 const Field& k2,
221 const Field& k3,
222 const Field& k4,
223 const Real& dt) const
224 {
225 const Index total = K_ * Np_;
226 Real* pu = u.data().getData();
227 const Real* p1 = k1.data().getData();
228 const Real* p2 = k2.data().getData();
229 const Real* p3 = k3.data().getData();
230 const Real* p4 = k4.data().getData();
231 const Real sixth = Real(1) / Real(6);
232 for (Index i = 0; i < total; ++i)
233 {
234 pu[i] += dt * sixth * (p1[i] + Real(2)*p2[i] + Real(2)*p3[i] + p4[i]);
235 }
236 }
237
238 // RHS operator (function)
239 RHSFunc rhs_;
240
241 // number of cells and nodes
242 Index K_, Np_;
243
244 // the four stages
245 Field k1_, k2_, k3_, k4_;
246 Field tmp_;
247
248 // callback function
249 Callback callback_;
250
251 // diagnostics variables, updated by step()
252 Index lastStepCount_{ 0 };
253 Real currentTime_{ 0 };
254};

257// low storage explicit five stage Runge-Kutta
258// see: https://ntrs.nasa.gov/api/citations/19940028444/downloads/19940028444.pdf
259template<typename Real = double,
260 typename Device = TNL::Devices::Host,
261 typename Index = int>
262class LSERK : Integrator< Real, Device, Index > // inherit protected methods
263{
264public:
265 using Field = FieldVector<Real, Device, Index>;
266 using RHSFunc = std::function<void(const Field&, Field&, const Real&)>;
267 using Callback = StepCallback<Real, Device, Index>;
268
269 // op – spatial residual (must outlive this integrator)
270 // K, Np – element count and nodes-per-element (needed to size scratch)
271 // callback – optional: called after every step, return false to stop early
272 explicit LSERK(RHSFunc rhs,
273 Index K, Index Np,
274 Callback callback = nullptr)
275 : rhs_(std::move(rhs))
276 , K_(K), Np_(Np)
277 , callback_(std::move(callback))
278 , du_(K, Np)
279 , tmp_(K, Np) {}
280
281 // single step
282 void step(Field& u, Real dt, Real t_in) override
283 {
284 if (dt <= Real(0))
285 throw std::invalid_argument("DG::Integrator::step: dt must be > 0");
286
287 // p_1 = u
288 Real* du_data = du_.data().getData();
289 Real* u_data = u.data().getData();
290 Real* tmp_data = tmp_.data().getData();
291
292
293 for (int i = 0; i < 5; i++)
294 {
295 // tmp = L(u, t_n + c_i dt)
296 rhs_(u, tmp_, t_in + c_[i] * dt);
297
298 // tmp = dt * tmp
299 this -> scaleField_(tmp_, dt);
300
301 // du_ = a_i * du_ + tmp
302 this -> addScaled_(tmp_, du_, a_[i], du_);
303
304 // u = u + b_i * du_
305 this -> addScaled_(u, du_, b_[i], u);
306 }
307 }

318

320 {
321 // clip the last step to reach t_end exactly
322 Real dt_actual = std::min(dt, t_end - t);
323
324 step(u, dt_actual, t);
325 t += dt_actual;
326 nstep += 1;
327
328 // Invoke optional callback; stop early if it returns false
329 if (callback_ && !callback_(t, u, nstep))
330 break;
331 }
332
333 return nstep;
334 }

344 Index numSteps() const override { return lastStepCount_; }

393template<typename Real = double,
394 typename Device = TNL::Devices::Host,

396class SSPRK : Integrator< Real, Device, Index > // inherit protected methods
397{
398public:
399 using Field = FieldVector<Real, Device, Index>;
400 using RHSFunc = std::function<void(const Field&, Field&, const Real&)>;
401 using Callback = StepCallback<Real, Device, Index>;
402
403 // op – spatial residual (must outlive this integrator)
404 // K, Np – element count and nodes-per-element (needed to size scratch)
405 // callback – optional: called after every step, return false to stop early
406 explicit SSPRK(RHSFunc rhs,
407 Index K, Index Np,
408 Callback callback = nullptr)
409 : rhs_(std::move(rhs)),
410 K_(K), Np_(Np),
411 callback_(std::move(callback)),
412 k1_(K, Np), k2_(K, Np),
413 tmp_(K, Np) {}
414
415 // single step
416 void step(Field& u, Real dt, Real t_in) override
417 {
418 if (dt <= Real(0))
419 throw std::invalid_argument("DG::Integrator::step: dt must be > 0");
420
421 // k1 = u^n + 1/2 dt L(u^n, t_n)
422 // tmp = L(u^n, t_n)
423 rhs_(u, tmp_, t_in);
424 // tmp = 1/2 dt tmp
425 this -> scaleField_(tmp_, 0.5 * dt);
426 // k1 = tmp + u^n
427 this -> addScaled_(tmp_, u, 1.0, k1_);
428
429 // k2 = k1 + 1/2 dt L(k1, t_n + 1/2 dt)
430 // tmp = L(k1, t_n + 1/2 dt)
431 rhs_(k1_, tmp_, t_in + 0.5 * dt);
432 // tmp = 1/2 tmp
433 this -> scaleField_(tmp_, 0.5 * dt);
434 // k2 = tmp + u^n
435 this -> addScaled_(tmp_, k1_, 1.0, k2_);
436
437 // k3 = 2/3 u^n + 1/2 k2 + 1/6 L(k2, t^n + dt)
438 // (k3 = k1) to save storage
439 // tmp = L(k2, t^n + dt)
440 rhs_(k2_, tmp_, t_in + dt);
441 // tmp = 1/6 dt tmp
442 this -> scaleField_(tmp_, Real(1.0/6.0) * dt);
443 // tmp = tmp + 1/3 k2
444 this -> addScaled_(tmp_, k2_, Real(1.0/3.0), tmp_);
445 // k3 = tmp + 2/3 u^n
446 this -> addScaled_(tmp_, u, Real(2.0/3.0), k1_);
447
448 // u^n+1 = k3 + 1/2 dt L(k3, t^n + 1/2 dt)
449 // tmp = L(k3, t^n + 1/2 dt)
450 rhs_(k1_, tmp_, t_in + 0.5 * dt);
451 // tmp = 1/2 dt tmp
452 this -> scaleField_(tmp_, 0.5 * dt);
453 // u = tmp + k3
454 this -> addScaled_(tmp_, k1_, 1.0, u);
455
456 }
