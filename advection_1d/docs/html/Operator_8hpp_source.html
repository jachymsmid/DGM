68 void computeRHS(const Field& u, Field& res, const Real& time) const
69 {
70 const Index K = mesh_.numElements();
71 const Index Np = ref_.numDOF();
72 const auto& Dr = ref_.Dr();
73 const auto& LIFT = ref_.LIFT();
74
75
76 std::vector<Real> vol(Np), flocal(Np), fluxJump(2);
77
78 // for every mesh element
79 for (Index k = 0; k < K; ++k)
80 {
81 // create cell arrays
82 const Real* uk = u.elementPtr(k);
83 Real* rk = res.elementPtr(k);
84
85 // physical flux at all nodes of the element
86 for (Index i = 0; i < Np; ++i)
87 {
88 flocal[i] = physFlux_(uk[i]);
89 }
90
91 // --- volume term ---
92 // vol = Dr * flocal
93 for (Index i = 0; i < Np; ++i)
94 {
95 Real s = 0;
96 for (Index j = 0; j < Np; ++j)
97 {
98 s += Dr(i,j) * flocal[j];
99 }
100 vol[i] = s;
101 }
102
103 // --- surface term ---
104 // left face
105 {
106 // current element's value at left face
107 Real u_int = physFlux_(uk[0]);
108 // neighbour's face value
109 Real u_ext;
110 // BC
111 if (mesh_.isBoundaryFace(k))
112 {
113 // periodic
114 u_ext = physFlux_(u.elementPtr(K-1)[Np-1]);
115 // ghost cell
116 // u_ext = u_int;
117 }
118 else
119 {
120 u_ext = physFlux_(u.elementPtr(mesh_.leftCellOfFace(k))[Np-1]);
121 }
122 // fluxJump[0] = fStar
123 fluxJump[0] = flux_.compute(u_ext, u_int, mesh_.leftNormal());
124 }
125
126 // right face
127 {
128 // current element's value at right face
129 Real u_int = physFlux_(uk[Np-1]);
130 // neighbour's face value
131 Real u_ext;
132
133 // BC
134 if (mesh_.isBoundaryFace(k+1))
135 {
136 // periodic: right neighbour is first element's left end
137 u_ext = physFlux_(u.elementPtr(0)[0]);
138 // ghost cell: right neighbour takes its own value
139 // u_ext = u_int;
140 }
141 else
142 {
143 u_ext = physFlux_(u.elementPtr(mesh_.rightCellOfFace(k+1))[0]);
144 }
145 // numerical flux f*(u-, u+)
146 fluxJump[1] = flux_.compute(u_int, u_ext, mesh_.rightNormal());
147 }
148
149 // --- assemble rhs ---
150 // rhs = J^{-1} * (-vol + LIFT * fluxJump)
151 Real Jinv = Real(1) / mesh_.jacobian(k);
152 for (Index i = 0; i < Np; ++i)
153 {
154 Real lift = LIFT(i,0)*fluxJump[0] + LIFT(i,1)*fluxJump[1];
155 rk[i] = Jinv * ( vol[i] + lift);
156 }
157 }
158 }
